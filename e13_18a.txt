Zero MACRO X, T
	IF T EQ 'Last'
		IF TYPE X EQ BYTE
		MOV X[LENGTH X - 1], BYTE PTR 0
		ELSE
			IF TYPE X EQ WORD
			MOV X[2 * LENGTH X - 2], WORD PTR 0
			ELSE
				IF TYPE X EQ DWORD
				MOV X[4 * LENGTH X - 4], DWORD PTR 0
				ELSE
					IF TYPE X EQ QWORD
					MOV X[8 * LENGTH X - 8], QWORD PTR 0
				ENDIF
			ENDIF
		ENDIF
	ELSE
	IF T EQ 'First'
		MOV X, TYPE X PTR 0
	ENDIF
ENDM
