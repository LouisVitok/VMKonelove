INCLUDE console.inc
.DATA
.CODE
START:	
	MOV EBP, ESP
	MOV ESI, 0
	MOV EDI, 0
L:	INCHAR AL	
	MOVZX EAX, AL
	CMP EAX, '('
	JE L1
	CMP EAX, ')'
	JE L2
	CMP EAX, ','
	JE L3
	PUSH EAX
	CMP ESI, 0
	JNE L1
	POP ESI
L1:	ADD ESI, 1
	JMP L
L2:	ADD EDI, 1
	CMP ESI, EDI
	JE L4
L3:	JMP L
L4:	POP ESI
	CMP ESP, EBP
	JE E
	POP EDI
	POP EAX
	CMP EAX, 'M'
	JE L5
	CMP ESI, EDI
	JB L8
	MOV ESI, EDI
	JMP L8
L5:	CMP ESI, EDI
	JA L8
	MOV ESI, EDI
L8:	PUSH ESI
	CMP DWORD PTR [ESP]+12, 'M'
	JNE L6
	POP EBX
	POP ESI
	POP EDI
	CMP ESI, EDI
	JA L7
	MOV ESI, EDI
L6:	CMP DWORD PTR [ESP]+12, 'm'
	JNE L
	POP EBX
	POP ESI
	POP EDI
	CMP ESI, EDI
	JB L7
	MOV ESI, EDI
L7:	PUSH ESI
	PUSH EBX
	JMP L
E:	OUTU ESI
EXIT
END START
