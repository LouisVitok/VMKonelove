INCLUDE console.inc
.DATA
	M=10
	N=9
	A DB M DUP(?)
	B DB N DUP(?)
	K DB ?
.CODE
	F PROC
		MOV EAX, 0
		MOV EBX, DWORD PTR [ESP + 8]
		MOV ECX, DWORD PTR [ESP + 12]
		MOV ESI, 0
		L:	MOV DL, BYTE PTR [EBX + ESI]
			CMP DL, BYTE PTR [ESP + 16]
			JNE L1
			INC EAX
		L1:	INC ESI
			LOOP L
		RET 12
	F ENDP
START:	
	MOV ECX, M
	MOV ESI, 0
L2:	ININT DL
	MOV BYTE PTR A[ESI], DL
	INC ESI
	LOOP L2
	MOV ECX, N
	MOV ESI, 0
L3:	ININT DL
	MOV BYTE PTR B[ESI], DL
	INC ESI
	LOOP L3
	ININT K
	MOVSX EBX, K
	PUSH EBX
	MOV EBX, N
	PUSH EBX
	LEA EBX, B
	PUSH EBX
	CALL F
	MOVSX EAX, AL
	PUSH EAX
	MOV EBX, M
	PUSH EBX
	LEA EBX, A
	PUSH EBX
	CALL F
	OUTI AL
E:	
EXIT
END START
