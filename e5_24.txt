INCLUDE console.inc
.DATA
.CODE
START:	
	MOV EBX, 2
	MOV ESI, 0
	ININT ECX
	MOV EBP, ECX
	CMP ECX, 2
	JB L1
	MOV EAX, ECX
	MOV EDX, 0
	ADD EAX, 1
	DIV EBX
	XCHG ECX, EAX
L:	MOV EDX, 0
	MOV EAX, EBP
	DIV ECX
	CMP EDX, 0
	JNE L3
	CMP ECX, EAX
	JE L2
	ADD ESI, 1
	LOOP L
	JMP L5
L2:	ADD ESI, 1
L3:	LOOP L
L5:	ADD ESI, 1
	OUTU ESI
	JMP E

L1:	OUTU 1
E:
EXIT
END START
