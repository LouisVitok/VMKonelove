INCLUDE console.inc
.DATA
	M=10
	N=9
	A DB M DUP(?)
	B DB N DUP(?)
.CODE
	PRINT PROC
	PUSH ECX
	PUSH ESI
	P:	OUTI BYTE PTR [ESI]
		INC ESI
		OUTCHAR ' '
		SUB ECX, 1
		CMP ECX, 0
		JNE P
	POP ESI
	POP ECX
	RET
	PRINT ENDP
START:	
	MOV ESI, 2
	MOV ECX, M
	MOV EDX, 0
L:	ININT AL
	MOV A[EDX], AL
	INC EDX
	LOOP L
	MOV ECX, N
	MOV EDX, 0
L1:	ININT AL
	MOV B[EDX], AL
	INC EDX
	LOOP L1
	MOV EDX, 0
	MOV EAX, N
	DEC EAX
	DIV ESI
	MOV ESI, M
	DEC ESI
	MOV BL, A[ESI]
	MOV BH, B[EAX]
	CMP BL, BH
	JE L2
	LEA ESI, B
	MOV ECX, N
	CALL PRINT
	JMP E
L2:	LEA ESI, A
	MOV ECX, M
	OUTCHAR ' '
	CALL PRINT
E:	
EXIT
END START
