INCLUDE console.inc
.DATA
	M=10
	N=9
	K=4
	A DB M DUP(?)
	B DB N DUP(?)
.CODE
	F PROC
		MOV EAX, 0
		MOV EBX, DWORD PTR [ESP + 8]
		MOV ECX, DWORD PTR [ESP + 12]
		MOV ESI, 0
		L:	MOV DL, BYTE PTR [EBX + ESI]
			CMP DL, BYTE PTR [ESP + 16]
			JNE L1
			INC EAX
		L1:	INC ESI
			LOOP L
		OUTI EAX
		RET 12
	F ENDP
START:	
	MOV EBX, K
	PUSH EBX
	MOV EBX, N
	PUSH EBX
	LEA ESI, B
	PUSH ESI
	CALL F
	OUTI AL
	MOVZX EAX, AL
	PUSH EAX
	MOV EBX, M
	PUSH EBX
	LEA ESI, A
	PUSH ESI
	CALL F
	OUTI AL
E:	
EXIT
END START
