INCLUDE console.inc
.DATA
	M=10
	N=9
	A DB M DUP(?)
	B DB N DUP(?)
	K DB ?
.CODE
	F PROC
		
		PUSH EBP
		MOV EBP, ESP
		PUSH EBX
		PUSH ECX
		PUSH EDX
		PUSH ESI
		MOV EAX, 0
		MOV EBX, DWORD PTR [EBP + 8]
		MOV ECX, DWORD PTR [EBP + 12]
		MOV ESI, 0
		L:	MOVZX EDX, BYTE PTR [EBX + ESI]
			CMP EDX, DWORD PTR [EBP + 16]
			JNE L1
			INC EAX
		L1:	INC ESI
			LOOP L
		POP ESI
		POP EDX
		POP ECX
		POP EBX
		POP EBP
		RET 12
	F ENDP
START:	
	MOV ECX, M
	MOV EAX, 0
L2:	ININT DL
	MOV A[EAX], DL
	INC EAX
	LOOP L2
	MOV ECX, N
	MOV EAX, 0
L3:	ININT DL
	MOV B[EAX], DL
	INC EAX
	LOOP L3
	ININT K
	MOVZX EBX, K
	PUSH EBX
	MOV EBX, N
	PUSH EBX
	LEA ESI, B
	PUSH ESI
	CALL F
	PUSH EAX
	MOV EBX, M
	PUSH EBX
	LEA ESI, A
	PUSH ESI
	CALL F
	OUTI AL
E:	
EXIT
END START
