INCLUDE console.inc
.DATA
.CODE
START:
	MOV EAX, 0h
	MOV ECX, 256
	INCHAR BL
	CMP BL, '-'
	JNE MPL
	MOV EDI, 1h
	JMP START
MPL:
	CMP BL, '+'
	JNE OK
	MOV EDI, 0
	JMP START
CI:
	INCHAR BL
	CMP BL, ' '
	JE VOD
OK:
	CMP BL, '0'
	JB ERR
	CMP BL, '9'
	JA ERR
	SUB BL, 48
	MOVZX EBP, BL
	MOV EBX, 10
	MUL EBX
	ADD EAX, EBP
	LOOP CI
VOD:
	CMP EDX, 0h
	JA OVER
	CMP EDI, 1h
	JE SMEN
	CMP EAX, 0h
	JL OVER
	OUTU EAX
	JMP FIN
SMEN:
	NEG EAX
	CMP EAX, 80000000h
	JL OVER
	CMP EAX, 0FFFFFFFh
	JG OVER
	OUTI EAX 
	JMP FIN
OVER: 
	OUTSTR 'overflow'
	JMP FIN
ERR:
	OUTSTR 'error'
	JMP FIN
FIN:
EXIT
END START
