INCLUDE console.inc
.DATA
.CODE
START:
	ININT ECX
	ININT EAX
	SUB ECX, 1
	MOV EDX, 0
	MOV ESI, EAX
	JECXZ L1
L:	ININT EBX
	CMP EBX, ESI
	JGE L2
	MOV ESI, EBX
L2:	LOOP L
L1:	CMP ESI, 0
	JE L3
	CMP EAX, ESI
	JE L4
	CMP EAX, 0
	JL L5
	IDIV ESI
	OUTI EAX
	JMP E
L3:	OUTSTR 'ERROR'
	JMP E
L4:	OUTU 1
	JMP E
L5:	OUTU 0
E:
EXIT
END START
