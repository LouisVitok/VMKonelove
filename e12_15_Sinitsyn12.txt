ZERO PROC
	CLD
	MOV EBP, ESP
	PUSH EBP
	PUSH ECX
	PUSH ESI
	PUSH EBX
	MOV ECX, DWORD PTR [EBP + 8]
	MOV ESI, DWORD PTR [EBP + 4]
	MOV EBX, 0
	L:	CMP DWORD PTR [ESI + EBX], 0
		JE L1
		ADD EBX, 4
		LOOP L
		JMP L2
	L1:	MOV EAX, -1
		JMP E
	L2:	MOV EAX, 0
	E:	
	POP EBX
	POP ESI
	POP ECX	
	POP EBP
	RET 8	
ZERO ENDP
